  ```bash
  # RNPT-02: Authoritative list of phases that require write access to the workdir.
  # Phase code MUST NOT pass a hard-coded "true"/"false" to invoke_agent; it must
  # call `phase_is_writable "<phase-name>"` instead. To add a new writable phase
  # (e.g. a future `fix` phase), append its name to this array.
  WRITABLE_PHASES=(execute execute-retry fix)

  # phase_is_writable <phase-name> → prints "true" or "false" on stdout.
  # Returns 0 on a known writable phase, 0 on a known text phase, 1 on unknown.
  # Callers pass the result directly as the 4th positional arg to invoke_agent.
  phase_is_writable() {
    local phase_name="${1:?phase_is_writable requires a phase name}"
    local candidate
    for candidate in "${WRITABLE_PHASES[@]}"; do
      if [[ "$phase_name" == "$candidate" ]]; then
        printf '%s' "true"
        return 0
      fi
    done
    printf '%s' "false"
    return 0
  }
  ```

  Constraints:
  - MUST NOT change `invoke_claude` / `invoke_codex` signatures or behavior (Phase 6 landed them; this plan only adds a helper).
  - MUST NOT introduce side effects at source time — library must remain source-safe (CORE-01).
  - Use `printf '%s'` (no trailing newline) so callers can inline-substitute: `invoke_agent "$AGENT" ... "$(phase_is_writable compose)"`.
  - Do NOT include "bounce" or "compose" or "review" in `WRITABLE_PHASES` — those are text phases (falling through the loop returns "false" correctly).
  - Do NOT return non-zero for "unknown phase" even though the docstring mentions it — bash `set -euo pipefail` in sourced scripts would make unknown-phase errors fatal at substitution time. Fail-safe to "false" instead (matches Phase 6's "default to safer posture" decision D-06 from the STATE.md decision log).
  </action>
